#include <cstdio>

using namespace std;

const int M = 1e5 + 10;

int n, k, arr[M];
int main() {
#ifndef ONLINE_JUDGE
    freopen("test.in", "r", stdin);
    freopen("test.out", "w", stdout);
#endif
    scanf("%d%d", &n, &k);
    for (int i = 0; i < n; i++) {
        scanf("%d", arr + i);
    }
    int tmp = arr[k - 1];
    for (int i = k; i < n; i++) {
        if (arr[i] != tmp) {
            printf("-1");
            return 0;
        }
    }
    for (int i = k - 2; i >= 0; i--) {
        if (arr[i] != tmp) {
            printf("%d", i + 1);
            return 0;
        }
    }
    printf("0");
    return 0;
}
